큐를 이용한 방식과 idx를 이용한 방식 두가지를 이용하여 해결하였다.

```
package implemention;

import java.util.*;
import java.util.stream.Collectors;

public class b11866 {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String[] str = sc.nextLine().split(" ");
        int N = Integer.parseInt(str[0]); // K번째 사람을 제거
        int K = Integer.parseInt(str[1]); // N명의 사람이 모두 제거될때까지

        Queue<Integer> queue = new LinkedList<Integer>();
        for (int i = 1; i <= N; i++) {
            queue.add(i);
        }
        List<String> result = new ArrayList<>();
        while(!queue.isEmpty()) {
            for (int i = 1; i < K; i++) {
                queue.add(queue.poll());
            }
            result.add(String.valueOf(queue.poll()));
        }
        System.out.println(result.stream()
                .collect(Collectors.joining(", ", "<" ,">")));
    }

}

```
package implemention;

import java.util.*;
import java.util.stream.Collectors;

public class b11866 {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String[] str = sc.nextLine().split(" ");
        int N = Integer.parseInt(str[0]); // K번째 사람을 제거
        int K = Integer.parseInt(str[1]) - 1; // N명의 사람이 모두 제거될때까지

        List<String> list = new ArrayList<>();
        for (int i = 1; i <= N; i++) {
            list.add(String.valueOf(i));
        }
        List<String> result = new ArrayList<>();
        int idx = 0;
        while (!list.isEmpty()) {
            idx = (idx + K) % list.size();
            result.add(list.get(idx));
            list.remove(idx);
        }


        System.out.println(result.stream()
                .collect(Collectors.joining(", ", "<" ,">")));
    }

}
